7-1 单链表按值操作 (50分)

对值递增有序的单链表进行以下操作：若表中存在值为x的结点，则将它从表中删除；否则，就往表中插入一个值为x的结点，并保持表值递增有序的性质不变（假设表中没有值相同的元素）。处理后若为空表则不输出。

输入格式:

每组数据包括3行，第一行表示单链表的长度n（0<=n<50）；第二行表示单链表的所有元素；第三行表示x值。

输出格式:

输出执行操作后的单链表，元素之间用一个空格分隔。

输入样例:

在这里给出一组输入。例如：

5
1 3 5 7 9
3
5
1 3 5 7 9
4

输出样例:

在这里给出相应的输出。例如：

1 5 7 9
1 3 4 5 7 9

#include<iostream>
using namespace std;
typedef struct LNode //创建单链表指针结构体（结构指针）
{
	int data; //结点的数据域
	struct LNode *next; //结点的指针域
}LNode,*LinkList;
void Creat(LinkList &L,int n) //创建单链表函数
{
	int i;
	LNode *p,*r; //创建指针
	L=new LNode; //使用尾插法
	L->next=NULL; //先建立一个带头结点的空链表
	r=L; //尾指针r指向头结点
	for(i=0;i<n;i++)
	{
		p=new LNode; //生成新结点
		cin>>p->data; //输入元素值赋给新结点*p的数据域
		p->next=NULL;r->next=p; //将新结点*p插入尾结点*r之后
		r=p; //r指向新的尾结点*p
	}
}
void Change(LinkList &L,int x) //主功能函数
{
	LNode *p,*q,*s; //创建指针
	q=L; //头结点赋给q
	p=q->next; //p指向q的指针域
	while(p && p->data<x) //如果p节点存在且p的数据域比目标数x小
	{
		q=p; //p结点赋给q结点
		p=p->next; //继续遍历单链表
	}
	if(p && p->data==x) //如果单链表元素中存在与x等值的元素
	{
		q->next=p->next; //p的指针域赋给q的指针域，相当于跳过p的数据域
		delete p; //删除p结点
	}
	else
	{
		s=new LNode;
		s->data=x;
		s->next=p;
		q->next=s; //以上三行为插入数据操作
	}
}
void Print(LinkList L) //Print()函数，依序输出单链表
{
	LNode *p=L->next;
	if(p==NULL)
	{
		return; //若表为空，退出函数执行
	}
	while(p->next) //若p的指针域不为空（即除单链表最后一个元素外），执行循环
	{
		cout<<p->data<<" "; //打印单链表元素（除单链表最后一个元素外）
		p=p->next; //指针移动
	}
	cout<<p->data<<endl; //打印单链表的最后一个元素
}
int main()
{
	int n,x;
	while(cin>>n) //循环输入单链表长度n
	{
		LinkList L; //初始化单链表
		Creat(L,n); //创建单链表
		cin>>x; //输入目标值x
		Change(L,x); //调用Change()函数，依照题意对单链表进行操作
		Print(L); //输出经过操作的单链表
	}
	return 0;
}

7-2 单链表删除相同值 (50分)

单链表A是非递减有序数列，编写算法，删除表中值相同的多余节点，可能有多组相同值。

输入格式:

输入包含多组测试数据，每组测试数据第一行是一个n，表示有n个数输入，第二行是n个数，当n为零时结束(n>=0)。

输出格式:

每组输出占一行，输出删除相同值后的列表，每两个数据之间有一个空格。

输入样例:

在这里给出一组输入。例如：

5
1 2 2 3 3
6
1 1 1 1 1 1
6
1 2 3 4 5 6
0

输出样例:

在这里给出相应的输出。例如：

1 2 3
1
1 2 3 4 5 6

#include<iostream>
using namespace std;
typedef struct LNode //创建单链表指针结构体（结构指针）
{
	int data; //结点的数据域
	struct LNode *next; //结点的指针域
}LNode,*LinkList;
void Creat(LinkList &L,int n) //创建单链表函数
{
	int i;
	LNode *p,*r; //创建指针
	L=new LNode; //使用尾插法
	L->next=NULL; //先建立一个带头结点的空链表
	r=L; //尾指针r指向头结点
	for(i=0;i<n;i++)
	{
		p=new LNode; //生成新结点
		cin>>p->data; //输入元素值赋给新结点*p的数据域
		p->next=NULL; r->next=p; //将新结点*p插入尾结点*r之后
		r=p; //r指向新的尾结点*p
	}
}
void Delete(LinkList &L) //主功能函数
{
	LNode *p,*q,*s; //创建指针
	p=L->next; //头结点的指针域赋给p
	q=p->next; //p的指针域赋给q
	while(p->next) //若p的指针域不为空，执行循环
	{
		if(p->data==q->data) //如果p的数据域与q的数据域等值
		{
			s=q; //q结点赋给s结点
			p->next=q->next; //q的指针域赋给p的指针域
			q=q->next; //q的指针域赋给q（即跳过q的数据域）
			delete s; //删除s结点
		}
		else //其他情况，指针继续移动，查找重复的单链表元素
		{
			p=p->next;
			q=q->next;
		}
	}
}
void Print(LinkList L) //Print()函数，依序输出单链表
{
	LNode *p=L->next;
	if(p==NULL)
	{
		return; //若表为空，退出函数执行
	}
	while(p->next) //若p的指针域不为空（即除单链表最后一个元素外），执行循环
	{
		cout<<p->data<<" "; //打印单链表元素（除单链表最后一个元素外）
		p=p->next; //指针移动
	}
	cout<<p->data<<endl; //打印单链表的最后一个元素
}
int main()
{
	int n;
	while(cin>>n) //循环输入单链表长度n
	{
		LinkList L; //初始化单链表
		if(n==0)
		{
			return 0; //如果初始表长n为0，结束程序
		}
		else //初始表长不为0
		{
			Creat(L,n); //创建单链表
			Delete(L); //调用Delete()函数，依照题意对单链表进行操作
			Print(L); //输出经过操作的单链表
		}
	}
	return 0;
}
